const sliderInner1 = $(".slider-inner1");
const slider1 = $(".slider1");

let currentIndex1 = 0;
let sliderHeight1 = slider1.height();
let sliderCount1 = slider1.length;
setInterval(function(){
    // if(currentIndex1 < sliderCount1 ){
    //     currentIndex1++;
    // } else {
    //     currentIndex1 = 0;
    // }
    (currentIndex1 < sliderCount1 -1 ) ? currentIndex1++ : currentIndex1 = 0;

    let sliderPosition1 = currentIndex1 * -sliderHeight1;
    sliderInner1.animate({top:sliderPosition1}, 400);
}, 3000);

const sliderInner2 = $(".slider-inner2");
